import os
import math
import requests
from flask import Flask, render_template, jsonify
from collections import defaultdict

app = Flask(__name__)

# Configuration
API_KEY = os.environ.get('ODDS_API_KEY')
if not API_KEY:
    raise ValueError("Please set ODDS_API_KEY environment variable")

ODDS_API_URL = "https://api.the-odds-api.com/v4/sports/{sport}/odds/"
SPORTS = ['americanfootball_nfl', 'basketball_nba', 'baseball_mlb']  # Add more as needed

# Simulated team power ratings (will be populated dynamically)
team_ratings = defaultdict(lambda: 50.0)  # default rating 50

def update_team_ratings_from_games(games):
    """
    Update team ratings based on game outcomes (if available) - here we simulate
    by slightly adjusting based on recent performance.
    In a real system, you'd use historical data and ML.
    """
    # For demo, we just ensure teams exist; ratings stay constant for simplicity
    for game in games:
        home_team = game['home_team']
        away_team = game['away_team']
        # Initialize if not present
        if home_team not in team_ratings:
            team_ratings[home_team] = 50.0
        if away_team not in team_ratings:
            team_ratings[away_team] = 50.0

def predict_winner_probability(home_team, away_team):
    """
    AI model: logistic function based on team ratings and home advantage.
    Returns probability that home team wins.
    """
    rating_home = team_ratings[home_team]
    rating_away = team_ratings[away_team]
    home_advantage = 5.0  # constant boost for home team
    rating_diff = rating_home - rating_away + home_advantage
    # Logistic function: 1 / (1 + exp(-rating_diff / scale))
    scale = 10.0  # scales the rating difference to probability
    prob_home = 1.0 / (1.0 + math.exp(-rating_diff / scale))
    return prob_home

def implied_probability(american_odds):
    """
    Convert American odds to implied probability.
    Positive odds: 100 / (odds + 100)
    Negative odds: -odds / (-odds + 100)
    """
    if american_odds > 0:
        return 100 / (american_odds + 100)
    else:
        return -american_odds / (-american_odds + 100)

def fetch_games(sport):
    """Fetch upcoming games from The Odds API for a given sport."""
    params = {
        'apiKey': API_KEY,
        'regions': 'us',  # US odds
        'markets': 'h2h',  # moneyline (head-to-head)
        'oddsFormat': 'american',
        'dateFormat': 'iso'
    }
    url = ODDS_API_URL.format(sport=sport)
    response = requests.get(url, params=params)
    if response.status_code != 200:
        print(f"Error fetching {sport}: {response.status_code}")
        return []
    return response.json()

def analyze_games(games):
    """
    For each game, compute model probability and compare with market.
    Return list of value bets.
    """
    recommendations = []
    for game in games:
        home_team = game['home_team']
        away_team = game['away_team']
        # Get moneyline odds from the first bookmaker (simplified)
        if not game['bookmakers']:
            continue
        bookmaker = game['bookmakers'][0]
        outcomes = bookmaker['markets'][0]['outcomes']
        # Find home and away odds
        home_odds = None
        away_odds = None
        for outcome in outcomes:
            if outcome['name'] == home_team:
                home_odds = outcome['price']
            elif outcome['name'] == away_team:
                away_odds = outcome['price']
        if home_odds is None or away_odds is None:
            continue

        # Model probability
        prob_home = predict_winner_probability(home_team, away_team)
        prob_away = 1 - prob_home

        # Market implied probabilities
        imp_home = implied_probability(home_odds)
        imp_away = implied_probability(away_odds)

        # Check for value (model > market)
        if prob_home > imp_home:
            recommendations.append({
                'sport': game['sport_title'],
                'game': f"{away_team} @ {home_team}",
                'bet': home_team,
                'model_prob': round(prob_home, 3),
                'market_prob': round(imp_home, 3),
                'odds': home_odds,
                'value': round(prob_home - imp_home, 3)
            })
        if prob_away > imp_away:
            recommendations.append({
                'sport': game['sport_title'],
                'game': f"{away_team} @ {home_team}",
                'bet': away_team,
                'model_prob': round(prob_away, 3),
                'market_prob': round(imp_away, 3),
                'odds': away_odds,
                'value': round(prob_away - imp_away, 3)
            })
    return recommendations

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/api/recommendations')
def get_recommendations():
    all_recommendations = []
    for sport in SPORTS:
        games = fetch_games(sport)
        update_team_ratings_from_games(games)
        recommendations = analyze_games(games)
        all_recommendations.extend(recommendations)
    # Sort by value (highest first)
    all_recommendations.sort(key=lambda x: x['value'], reverse=True)
    return jsonify(all_recommendations)

if __name__ == '__main__':
    app.run(debug=True)
